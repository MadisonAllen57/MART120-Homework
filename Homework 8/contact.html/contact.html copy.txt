  <form id="contactForm" class="box" novalidate>
    <label for="name">Name:</label><br>
    <input id="name" name="name" type="text" required><br>
    <small id="nameHelp" class="help">Name is required.</small>
    <br><br>

    <label for="email">Email:</label><br>
    <input id="email" name="email" type="email" required><br>
    <small id="emailHelp" class="help">Email is required.</small>
    <br><br>

    <label>Accept Terms of Agreement?</label><br>
    <input type="radio" id="yes" name="terms" value="yes" required>
    <label for="yes">Yes</label>
    <input type="radio" id="no" name="terms" value="no">
    <label for="no">No</label><br>
    <small id="termsHelp" class="help">Please pick Yes or No.</small>
    <br><br>

    <button type="submit" class="btn btn-primary">Send</button>
  </form>


  <script>
    const form = document.getElementById('contactForm');
    const nameInput = document.getElementById('name');
    const emailInput = document.getElementById('email');
    const yes = document.getElementById('yes');
    const no = document.getElementById('no');

    const nameHelp = document.getElementById('nameHelp');
    const emailHelp = document.getElementById('emailHelp');
    const termsHelp = document.getElementById('termsHelp');

    function clearErrors() {
      [nameInput, emailInput].forEach(el => el.classList.remove('error'));
      document.getElementById('yes').parentElement.classList.remove('error');
      nameHelp.style.display = 'none';
      emailHelp.style.display = 'none';
      termsHelp.style.display = 'none';
    }

    form.addEventListener('submit', function (e) {
      clearErrors();
      let firstBad = null;

      if (nameInput.value.trim() === '') {
        nameInput.classList.add('error');
        nameHelp.style.display = 'inline';
        if (!firstBad) firstBad = nameInput;
      }

      if (emailInput.value.trim() === '') {
        emailInput.classList.add('error');
        emailHelp.style.display = 'inline';
        if (!firstBad) firstBad = emailInput;
      }

      if (!yes.checked && !no.checked) {
        
        yes.parentElement.classList.add('error');
        termsHelp.style.display = 'inline';
        if (!firstBad) firstBad = yes;
      }

      if (firstBad) {
        e.preventDefault();
        firstBad.focus();
      }
    });

   
    nameInput.addEventListener('input', () => { nameInput.classList.remove('error'); nameHelp.style.display = 'none'; });
    emailInput.addEventListener('input', () => { emailInput.classList.remove('error'); emailHelp.style.display = 'none'; });
    yes.addEventListener('change', () => { yes.parentElement.classList.remove('error'); termsHelp.style.display = 'none'; });
    no.addEventListener('change', () => { yes.parentElement.classList.remove('error'); termsHelp.style.display = 'none'; });
  </script>
